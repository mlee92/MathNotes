\begin{definition}
	For functions $f, g \colon \mathbb{R} \to \mathbb{C}$, their Hermitian form (i.e. inner product) is
	{\center
		$\langle f, g \rangle =  \int_{-\infty}^{\infty}{\bar{f(x)}g(x)}{\mathrm{dx}}$ \\

	}
	
\end{definition}

\begin{definition}
	The Fourier Transofrm is 
	{\center
		$\widetilde{f}{(\omega)} = \langle{ \frac{1}{\sqrt{2\pi}} {e^{-i\omega x}}, f} = $ \\
		$ =  {\langle \frac{1}{\sqrt{2\pi}} \int_{-\infty}^{\infty}{e^{-i\omega x}}{\mathrm{dx}}}\rangle$ \\
	}
\end{definition}

\section{Properties of the Fourier Transform}

\subsection{Linearity}

The Fourier transform of a sum of functions is equal to the sum of the Fourier transforms of the functions. Since we can obviously factor out constants, the Fourier Transofrm is a linear operator. \\
For functions $f, g \colon \mathbb{R} \to \mathbb{C}$, \\
{\center
	$\mathbb{F}\lbrace f, g \rbrace (\omega) = \frac{1}{2\pi} \int_{-\infty}^{\infty}{exp(-i\omega x))(f(x) + g(x))}{\mathrm{dx}} $ \\
	$= \frac{1}{2\pi} \int_{-\infty}^{\infty}{exp(-i\omega x)}f(x){\mathrm{dx}} + \frac{1}{2\pi} \int_{-\infty}^{\infty}{exp(-i\omega x) g(x)}{\mathrm{dx}} $ \\
	$ = \widetilde{f}(\omega) + \widetilde{g}(\omega) $\\
}
